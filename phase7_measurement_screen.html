lib/features/measure/domain/measurement_pin.dart（新規作成） 
lib/features/measure/domain/measurement_pin.dart/// 地図上のピン1点を表すデータクラス。
/// MeasurementNotifier の pins リストで管理し、
/// MeasurementScreen と MeasurementListScreen の両方が参照する。
enum PinStatus {
  measuring,   // 赤: 測定中 / 打点直後 / エラー
  measured,    // 黄: CSV ローカル保存済み（アップロード待ちまたは処理中）
  completed,   // 緑: API が completed を返した（推定完了）
}

class MeasurementPin {
  final int       index;      // 1始まりの番号（表示用）
  final double    latitude;
  final double    longitude;
  final PinStatus status;
  final double?   cec;        // 推定完了後に値が入る
  final bool      hasError;

  const MeasurementPin({
    required this.index,
    required this.latitude,
    required this.longitude,
    this.status   = PinStatus.measuring,
    this.cec,
    this.hasError = false,
  });

  /// status を変えた新しい MeasurementPin を返す（immutable 更新）
  MeasurementPin copyWith({
    PinStatus? status,
    double?    cec,
    bool?      hasError,
    double?    latitude,
    double?    longitude,
  }) => MeasurementPin(
    index:     index,
    latitude:  latitude  ?? this.latitude,
    longitude: longitude ?? this.longitude,
    status:    status    ?? this.status,
    cec:       cec       ?? this.cec,
    hasError:  hasError  ?? this.hasError,
  );
}

現行コードからロジックを移植（メソッドごとに確認しながら）

measurement_notifier.dart に追記

以下の順番で1メソッドずつ移植し、 各移植後に flutter analyze でエラーがないことを確認してください。

① connectSensor() — USB シリアル接続 + Recall

  • 現行の Step1 の「接続」ボタン onPressed のロジックを移植
  • UsbSerial のオープン → "condition recall 0" 送信 → タイムアウト 20秒
  • 完了後: isSensorConnected = true → notifyListeners()
  • 失敗後: isSensorConnected = false、ログに追記

② startBg() — BG 測定コマンド送信

  • 現行の Step2 の「BG測定開始」ボタンのロジックを移植
  • コマンド: "null 1.0 0.5 0.1 1"（デフォルト値。設定値があればそれを使う）
  • 完了後: isBgDone = true → notifyListeners()

③ loadFarmCandidates() — GPS + 圃場候補絞り込み

  • 現在地を取得して GPS に近い順に圃場候補を返す（現行 Step3 のロジック）
  • GPS は Geolocator.getCurrentPosition
  • 結果を farmCandidates にセット → notifyListeners()

④ startMeasurement() — 測定実行（最も複雑）

  • 現在地（またはタップ補正位置）から PolygonUtil.containsLocation で圃場内外判定
  • outside の場合は例外 / エラーログを出して処理中断
  • 測定実行（CSV生成）→ ApplicationDocumentsDirectory に保存
  • ローカル保存完了後: ピンを赤→黄に更新 → notifyListeners()
  • アップロード（init → S3 PUT → complete）
  • アップロード完了後: ピンを黄→緑に更新 → notifyListeners()
  • アップロード失敗: PendingUploadStore に追加 → ピンは黄のまま
startMeasurement() の移植は段階的に行ってください。 一度にすべてを移植しようとすると型エラーや依存関係の問題が連鎖します。 以下の順番で少しずつ移植・確認を繰り返してください。
  1. 圃場内外判定のみ移植してテスト
  2. CSV 生成・ローカル保存を追加してテスト
  3. ピン状態更新を追加してテスト
  4. アップロードロジックを追加してテスト

class _StatusChip extends StatelessWidget {
  const _StatusChip({
    required this.label,    required this.subLabel,
    required this.isDone,   required this.isLoading,
    this.onTap,
  });
  final String       label, subLabel;
  final bool         isDone, isLoading;
  final VoidCallback? onTap;

  @override
  Widget build(BuildContext context) {
    final bg = isDone
        ? const Color(0xFF2E5C39)
        : onTap != null ? const Color(0xFF5A5A5A) : Colors.grey[300];
    final fg = isDone || onTap != null ? Colors.white : Colors.black38;

    return GestureDetector(
      onTap: onTap,
      child: Container(
        padding: const EdgeInsets.symmetric(horizontal: 10, vertical: 6),
        decoration: BoxDecoration(color: bg, borderRadius: BorderRadius.circular(20)),
        child: isLoading
            ? SizedBox(width: 16, height: 16,
                child: CircularProgressIndicator(strokeWidth: 2,
                    color: fg))
            : Column(mainAxisSize: MainAxisSize.min, children: [
                Text(isDone ? '✓ $label' : label,
                    style: TextStyle(color: fg, fontSize: 12, fontWeight: FontWeight.bold)),
                Text(subLabel,
                    style: TextStyle(color: fg.withValues(alpha: 0.8), fontSize: 10)),
              ]),
      ),
    );
  }
}

_MeasurementMap — 圃場ポリゴン + 現在地 + 測定ピンを描画class _MeasurementMap extends StatefulWidget {
  const _MeasurementMap({required this.notifier});
  final MeasurementNotifier notifier;
  @override
  State<_MeasurementMap> createState() => _MeasurementMapState();
}

class _MeasurementMapState extends State<_MeasurementMap> {
  GoogleMapController? _mapCtrl;

  @override
  Widget build(BuildContext context) {
    final n = widget.notifier;

    // 測定ピンを Marker Set に変換
    final markers = <Marker>{};

    // 現在地（青丸）
    if (n.currentLocation != null) {
      markers.add(Marker(
        markerId: const MarkerId('_current'),
        position: n.currentLocation!,
        icon: BitmapDescriptor.defaultMarkerWithHue(
            BitmapDescriptor.hueAzure),
        zIndex: 100,
      ));
    }

    // 測定ピン（状態に応じて色を変える）
    for (final pin in n.pins) {
      final hue = switch (pin.status) {
        PinStatus.measuring  => BitmapDescriptor.hueRed,
        PinStatus.measured   => BitmapDescriptor.hueYellow,
        PinStatus.completed  => BitmapDescriptor.hueGreen,
      };
      markers.add(Marker(
        markerId: MarkerId('pin_${pin.index}'),
        position: LatLng(pin.latitude, pin.longitude),
        icon: BitmapDescriptor.defaultMarkerWithHue(hue),
        infoWindow: InfoWindow(
          title: '${pin.index}' + (pin.cec != null ? ': ${pin.cec!.toStringAsFixed(1)}' : ''),
        ),
        // ピンのタップは地図上では無効（誤操作防止）。操作はリスト画面から
        onTap: () {},
      ));
    }

    // 圃場ポリゴン（赤枠）
    final polygons = <Polygon>{};
    // n.selectedFarmPolygon が LatLng リストで提供される前提。
    // 既存の圃場ポリゴン描画ロジックを流用してください。

    return GoogleMap(
      initialCameraPosition: CameraPosition(
        target: n.currentLocation ?? const LatLng(35.681236, 139.767125),
        zoom: 17,
      ),
      mapType: MapType.satellite,
      myLocationButtonEnabled: false,
      markers: markers,
      polygons: polygons,
      onMapCreated: (ctrl) => _mapCtrl = ctrl,
      // 地図タップ = 測定ピンの位置補正（onTap で currentLocation を上書き）
      onTap: (latLng) => n.overrideLocation(latLng),
    );
  }
}
圃場ポリゴンの描画ロジックについて。 現行の MeasurementSessionScreen で圃場境界を地図に描画しているコードがあるはずです。 grep -n "Polygon\|boundary_polygon\|boundaryPolygon" $(find lib/features/measure -name "*.dart") で確認して、同じロジックを _MeasurementMapState に移植してください。

lib/features/measure/presentation/measurement_list_screen.dartimport 'package:flutter/material.dart';
import 'package:provider/provider.dart';
import 'measurement_notifier.dart';
import '../domain/measurement_pin.dart';

class MeasurementListScreen extends StatefulWidget {
  const MeasurementListScreen({super.key});
  @override
  State<MeasurementListScreen> createState() => _MeasurementListScreenState();
}

class _MeasurementListScreenState extends State<MeasurementListScreen> {
  final _selected = <int>{};  // 選択中のピン index
  bool _showUnconfirmedOnly = false;

  @override
  Widget build(BuildContext context) {
    final n    = context.watch<MeasurementNotifier>();
    final pins = _showUnconfirmedOnly
        ? n.pins.where((p) => p.status != PinStatus.completed).toList()
        : n.pins;

    return Scaffold(
      appBar: AppBar(
        title: Text(n.selectedFarmName != null ? '測定リスト - ${n.selectedFarmName}' : '測定リスト'),
        backgroundColor: const Color(0xFF1A2318),
        foregroundColor: Colors.white,
        actions: [
          // 「未確定のみ」フィルタ
          TextButton(
            onPressed: () => setState(() => _showUnconfirmedOnly = !_showUnconfirmedOnly),
            child: Text(
              '未確定のみ',
              style: TextStyle(
                color: _showUnconfirmedOnly ? const Color(0xFF7DD3A8) : Colors.white70,
              ),
            ),
          ),
        ],
      ),
      body: Column(children: [

        // 件数表示
        Container(
          padding: const EdgeInsets.symmetric(horizontal: 16, vertical: 8),
          color: const Color(0xFFF5F5F0),
          child: Text('全 ${n.pins.length} 件',
              style: TextStyle(color: Colors.grey[700], fontSize: 13)),
        ),

        // ピンリスト
        Expanded(
          child: ListView.separated(
            itemCount: pins.length,
            separatorBuilder: (_, __) => const Divider(height: 1),
            itemBuilder: (ctx, i) {
              final pin = pins[i];
              final isError = pin.hasError;
              return CheckboxListTile(
                tileColor: isError ? const Color(0xFFFCE4E4) : null,
                value: _selected.contains(pin.index),
                onChanged: (v) => setState(() => v!
                    ? _selected.add(pin.index)
                    : _selected.remove(pin.index)),
                leading: CircleAvatar(
                  radius: 14,
                  backgroundColor: _pinColor(pin.status),
                  child: Text('${pin.index}',
                      style: const TextStyle(color: Colors.white, fontSize: 11)),
                ),
                title: Row(children: [
                  Text('CEC',  style: TextStyle(fontSize: 11, color: Colors.grey[600])),
                  const SizedBox(width: 4),
                  Text(pin.cec?.toStringAsFixed(1) ?? '—',
                      style: const TextStyle(fontWeight: FontWeight.bold)),
                ]),
                trailing: isError
                    ? const Text('エラー',
                        style: TextStyle(color: Colors.red, fontSize: 12))
                    : Text(_statusLabel(pin.status),
                        style: TextStyle(color: Colors.grey[600], fontSize: 12)),
              );
            },
          ),
        ),

        // 下部ボタン
        SafeArea(
          child: Padding(
            padding: const EdgeInsets.all(12),
            child: Row(children: [
              OutlinedButton(
                onPressed: _selected.isEmpty ? null : () {
                  for (final idx in _selected) n.deletePin(idx);
                  setState(() => _selected.clear());
                },
                child: const Text('選択を削除'),
              ),
              const SizedBox(width: 12),
              Expanded(
                child: ElevatedButton(
                  onPressed: _selected.isEmpty ? null : () {
                    // 位置修正: 選択中の1点の位置を地図に戻って補正する
                    if (_selected.length == 1) {
                      n.startPinCorrection(_selected.first);
                      Navigator.pop(context);
                    }
                  },
                  child: const Text('位置を修正'),
                ),
              ),
            ]),
          ),
        ),
      ]),
    );
  }

  Color _pinColor(PinStatus s) => switch (s) {
    PinStatus.measuring => const Color(0xFFC0392B),
    PinStatus.measured  => const Color(0xFFE67E22),
    PinStatus.completed => const Color(0xFF27AE60),
  };

  String _statusLabel(PinStatus s) => switch (s) {
    PinStatus.measuring => '測定中',
    PinStatus.measured  => '測定完了',
    PinStatus.completed => '推定完了',
  };
}
