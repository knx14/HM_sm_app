      "type": "work_log",

timeseries_result.dart  — API レスポンスを Dart オブジェクトにマッピング/// GET /api/farms/{id}/results/timeseries のレスポンス全体
class TimeseriesResult {
  final String              parameter;
  final String?             unit;
  final List<TimeseriesPoint> points;    // 日付昇順
  final List<WorkLogMark>    workLogs;  // グラフ上の縦線

  const TimeseriesResult({
    required this.parameter,
    this.unit,
    required this.points,
    required this.workLogs,
  });

  factory TimeseriesResult.fromJson(Map<String, dynamic> json) =>
      TimeseriesResult(
        parameter: json['parameter'] as String,
        unit:      json['unit']      as String?,
        points: (json['points'] as List)
            .map((e) => TimeseriesPoint.fromJson(e as Map<String, dynamic>))
            .toList(),
        workLogs: (json['work_logs'] as List)
            .map((e) => WorkLogMark.fromJson(e as Map<String, dynamic>))
            .toList(),
      );
}

/// 測定日ごとの集計値（avg / min / max / count）
class TimeseriesPoint {
  final String date;
  final double avg;
  final double min;
  final double max;
  final int    count;

  const TimeseriesPoint({
    required this.date, required this.avg,
    required this.min,  required this.max, required this.count,
  });

  factory TimeseriesPoint.fromJson(Map<String, dynamic> json) => TimeseriesPoint(
    date:  json['date']  as String,
    avg:   (json['avg']   as num).toDouble(),
    min:   (json['min']   as num).toDouble(),
    max:   (json['max']   as num).toDouble(),
    count: json['count'] as int,
  );
}

/// 折れ線グラフの縦線用（作業ログのタイミング）
class WorkLogMark {
  final String date;
  final String workType; // 'fertilization' | 'tillage' | etc.
  final String? title;

  const WorkLogMark({required this.date, required this.workType, this.title});

  factory WorkLogMark.fromJson(Map<String, dynamic> json) => WorkLogMark(
    date:     json['date']      as String,
    workType: json['work_type'] as String,
    title:    json['title']     as String?,
  );
}

タイムラインアイテムモデルを定義

lib/features/farms/domain/timeline_item.dart（新規作成） 
timeline_item.dart  — タイムライン API レスポンスの1アイテム（Union型で表現）/// items 配列の1要素の基底。type で判別する。
sealed class TimelineItem {
  final String date;
  final String type;
  const TimelineItem({required this.date, required this.type});

  factory TimelineItem.fromJson(Map<String, dynamic> json) {
    return switch (json['type'] as String) {
      'measurement' => MeasurementItem.fromJson(json),
      'work_log'    => WorkLogItem.fromJson(json),
      _             => throw ArgumentError('unknown type: ${json["type"]}'),
    };
  }
}

/// type = "measurement" のアイテム
final class MeasurementItem extends TimelineItem {
  final int                           countPoints;
  final Map<String, ParameterStat>    values;  // キー: 'CEC' | 'CaO' | 'K2O' | 'MgO'
  final double?                        deltaCec; // null = 最初の測定

  const MeasurementItem({
    required super.date,
    required this.countPoints,
    required this.values,
    this.deltaCec,
  }) : super(type: 'measurement');

  factory MeasurementItem.fromJson(Map<String, dynamic> json) {
    final valuesRaw = json['values'] as Map<String, dynamic>? ?? {};
    final delta     = json['delta']  as Map<String, dynamic>? ?? {};
    return MeasurementItem(
      date:        json['date']         as String,
      countPoints: json['count_points'] as int,
      values: valuesRaw.map((k, v) =>
          MapEntry(k, ParameterStat.fromJson(v as Map<String, dynamic>))),
      deltaCec: (delta['CEC'] as num?)?.toDouble(),
    );
  }
}

/// avg / min / max / unit を持つ集計値
class ParameterStat {
  final double avg, min, max;
  final String? unit;
  const ParameterStat({required this.avg, required this.min, required this.max, this.unit});
  factory ParameterStat.fromJson(Map<String, dynamic> j) => ParameterStat(
    avg:  (j['avg'] as num).toDouble(),
    min:  (j['min'] as num).toDouble(),
    max:  (j['max'] as num).toDouble(),
    unit:  j['unit'] as String?,
  );
}

/// type = "work_log" のアイテム
final class WorkLogItem extends TimelineItem {
  final String  workType;
  final String? title;
  final String? detail;
  final double? amountValue;
  final String? amountUnit;

  const WorkLogItem({
    required super.date,
    required this.workType,
    this.title, this.detail, this.amountValue, this.amountUnit,
  }) : super(type: 'work_log');

  factory WorkLogItem.fromJson(Map<String, dynamic> json) => WorkLogItem(
    date:        json['date']         as String,
    workType:    json['work_type']    as String,
    title:       json['title']        as String?,
    detail:      json['detail']       as String?,
    amountValue: (json['amount_value'] as num?)?.toDouble(),
    amountUnit:  json['amount_unit']  as String?,
  );
}
sealed class / final class について。 Dart 3.0+ の機能です。プロジェクトの pubspec.yaml の sdk: '>=3.0.0' であれば使えます。 古いバージョンの場合は abstract class + if (item is MeasurementItem) での判別に変えてください。 dart --version で確認できます。

ResultsRepository に追記する2メソッド  /// GET /api/farms/{farmId}/results/timeseries?parameter={param}
  /// 時系列タブ用: 測定日ごとの平均/min/max + 作業ログ縦線
  Future<TimeseriesResult> fetchTimeseries(
    int farmId, {
    String parameter = 'CEC',
  }) async {
    final res = await _dio.get(
      '/api/farms/$farmId/results/timeseries',
      queryParameters: {'parameter': parameter},
    );
    return TimeseriesResult.fromJson(res.data as Map<String, dynamic>);
  }

  /// GET /api/farms/{farmId}/timeline
  /// タイムラインタブ用: 測定結果 + 作業ログを日付降順でマージ
  Future<List<TimelineItem>> fetchTimeline(int farmId) async {
    final res  = await _dio.get('/api/farms/$farmId/timeline');
    final list = res.data['items'] as List;
    return list
        .map((e) => TimelineItem.fromJson(e as Map<String, dynamic>))
        .toList();
  }

timeseries_notifier.dart  — ChangeNotifier ベースの例import 'package:flutter/foundation.dart';
import '../../data/results_repository.dart';  // 実際のパスに合わせる
import '../../domain/timeseries_result.dart';

class TimeseriesNotifier extends ChangeNotifier {
  final int               _farmId;
  final ResultsRepository _repo;

  TimeseriesNotifier({required int farmId, required ResultsRepository repo})
      : _farmId = farmId, _repo = repo;

  // ── 公開状態 ─────────────────────────────────────────────
  String           selectedParam = 'CEC';
  TimeseriesResult? data;
  bool             isLoading = false;
  String?          error;

  /// 画面初期化時に呼ぶ
  Future<void> init() => _load();

  /// パラメータ切替（タブ切替ボタンから呼ぶ）
  Future<void> selectParam(String param) async {
    if (selectedParam == param) return;
    selectedParam = param;
    await _load();
  }

  Future<void> _load() async {
    isLoading = true;
    error     = null;
    notifyListeners();
    try {
      data = await _repo.fetchTimeseries(_farmId, parameter: selectedParam);
    } catch (e) {
      error = e.toString();
    } finally {
      isLoading = false;
      notifyListeners();
    }
  }
}

// ── fl_chart 折れ線グラフ ─────────────────────────────────────
class _TimeseriesChart extends StatelessWidget {
  const _TimeseriesChart({required this.data});
  final TimeseriesResult data;

  @override
  Widget build(BuildContext context) {
    final pts    = data.points;
    final labels = pts.map((p) => p.date.substring(5)).toList(); // "MM/DD"

    // avg の折れ線
    final avgSpots = pts.indexed
        .map(((i, p)) => FlSpot(i.toDouble(), p.avg))
        .toList();

    // min/max バンド（fl_chart の BetweenBarsData を使用）
    final minSpots = pts.indexed
        .map(((i, p)) => FlSpot(i.toDouble(), p.min))
        .toList();
    final maxSpots = pts.indexed
        .map(((i, p)) => FlSpot(i.toDouble(), p.max))
        .toList();

    // 作業ログ縦線 X 座標を特定（日付がpoints内に存在する場合）
    final dateIndex = {
      for (var i = 0; i < pts.length; i++) pts[i].date: i.toDouble()
    };
    final extraLines = data.workLogs
        .where((w) => dateIndex.containsKey(w.date))
        .map((w) => VerticalLine(
          x:          dateIndex[w.date]!,
          color:      Colors.orange.withValues(alpha: 0.7),
          strokeWidth: 1.5,
          dashArray:  [5, 3],
          label: VerticalLineLabel(
            show:  true,
            alignment: Alignment.topRight,
            style: const TextStyle(fontSize: 9, color: Colors.orange),
            labelResolver: (_) => w.workType == 'fertilization' ? '施肥'
                : w.workType == 'tillage' ? '耕うん' : '作業',
          ),
        ))
        .toList();

    return LineChart(
      LineChartData(
        extraLinesData: ExtraLinesData(verticalLines: extraLines),
        lineBarsData: [
          // max ライン（バンド上辺、非表示用）
          LineChartBarData(
            spots: maxSpots, color: Colors.transparent,
            dotData: const FlDotData(show: false),
          ),
          // min ライン（バンド下辺、非表示用）
          LineChartBarData(
            spots: minSpots, color: Colors.transparent,
            dotData: const FlDotData(show: false),
            belowBarData: BarAreaData(show: false),
          ),
          // avg ライン（メイン）
          LineChartBarData(
            spots: avgSpots,
            isCurved: true,
            color: Colors.green.shade700,
            barWidth: 2.5,
            dotData: FlDotData(
              getDotPainter: (_, __, ___, ____) => FlDotCirclePainter(
                radius: 4, color: Colors.green.shade700,
                strokeColor: Colors.white, strokeWidth: 1.5,
              ),
            ),
            belowBarData: BarAreaData(show: false),
          ),
        ],
        betweenBarsData: [
          BetweenBarsData(
            fromIndex: 0, toIndex: 1, // max と min の間をバンドで塗る
            color: Colors.green.shade100.withValues(alpha: 0.5),
          ),
        ],
        titlesData: FlTitlesData(
          bottomTitles: AxisTitles(
            sideTitles: SideTitles(
              showTitles: true,
              interval: 1,
              getTitlesWidget: (v, _) {
                final i = v.toInt();
                if (i < 0 || i >= labels.length) return const SizedBox.shrink();
                return Text(labels[i], style: const TextStyle(fontSize: 10));
              },
            ),
          ),
          topTitles:   const AxisTitles(sideTitles: SideTitles(showTitles: false)),
          rightTitles: const AxisTitles(sideTitles: SideTitles(showTitles: false)),
        ),
        gridData:    const FlGridData(show: true),
        borderData:  FlBorderData(show: false),
      ),
    );
  }
}
作業ログの縦線が points の日付と一致しない場合。 バックエンドの time series API は work_logs を測定日と独立して返します。 作業日が測定日のいずれとも一致しない場合（施肥日 = 測定しない日）は X 軸に整数インデックスを使っているため縦線が描けません。 その場合は extraLines に追加しないようにフィルタ済みです（dateIndex.containsKey）。 将来的に連続した日付 X 軸に対応する場合は X 軸を日付値（epoch）に変えてください。

class _TimelineView extends StatelessWidget {
  const _TimelineView();

  @override
  Widget build(BuildContext context) {
    final n = context.watch<TimelineNotifier>();

    if (n.isLoading) return const Center(child: CircularProgressIndicator());
    if (n.error != null)
      return Center(child: Text('エラー: ${n.error}'));
    if (n.items.isEmpty)
      return const Center(child: Text('測定データ・作業ログがありません'));

    return ListView.separated(
      padding: const EdgeInsets.all(12),
      itemCount: n.items.length,
      separatorBuilder: (_, __) => const SizedBox(height: 8),
      itemBuilder: (context, i) {
        final item = n.items[i];
        return switch (item) {
          MeasurementItem() => _MeasurementCard(item: item),
          WorkLogItem()     => _WorkLogCard(item: item),
        };
      },
    );
  }
}
