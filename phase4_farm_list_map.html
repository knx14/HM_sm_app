lib/models/farm_summary.dart  — 圃場一覧カードに必要なデータ/// 圃場一覧カードの1行分に必要なデータ。
/// GET /api/v1/farms レスポンスの各要素をマッピングする。
class FarmSummary {
  final int     id;
  final String  farmName;
  final String? cropType;
  final double? areaSqm;          // 面積（㎡）
  final String? lastMeasuredDate; // 最終測定日 "YYYY-MM-DD"
  final int?    lastPointCount;   // 最終測定の点数
  final double? avgCec;
  final double? avgCao;
  final double? avgK2o;
  final double? avgMgo;

  const FarmSummary({
    required this.id,
    required this.farmName,
    this.cropType,
    this.areaSqm,
    this.lastMeasuredDate,
    this.lastPointCount,
    this.avgCec,
    this.avgCao,
    this.avgK2o,
    this.avgMgo,
  });

  factory FarmSummary.fromJson(Map<String, dynamic> json) {
    final latest = json['latest_result'] as Map? ?? {};
    return FarmSummary(
      id:              json['id'] as int,
      farmName:        json['farm_name'] as String,
      cropType:        json['crop_type']       as String?,
      areaSqm:         (json['area_sqm']       as num?)?.toDouble(),
      lastMeasuredDate:latest['measurement_date'] as String?,
      lastPointCount:  latest['count_points']   as int?,
      avgCec:          (latest['avg_cec']  as num?)?.toDouble(),
      avgCao:          (latest['avg_cao']  as num?)?.toDouble(),
      avgK2o:          (latest['avg_k2o']  as num?)?.toDouble(),
      avgMgo:          (latest['avg_mgo']  as num?)?.toDouble(),
    );
  }
}

class MeasurementDate {
  final String date;   // "YYYY-MM-DD"
  final int    count;  // その日の測定点数

  const MeasurementDate({required this.date, required this.count});

  factory MeasurementDate.fromJson(Map<String, dynamic> json) =>

        count: json['count'] as int,
      );
}

既存 API クライアントに測定日一覧・マップデータ取得メソッドを追加

lib/api/farm_api_client.dart（または既存の相当ファイルを編集）

既存クラスを開き、以下の2メソッドを末尾に追記します。 既存の dio インスタンスとベースURL設定をそのまま流用します。 

追記するメソッド 2本（既存クラス内の末尾に追記）  /// GET /api/farms/{farmId}/results/dates
  /// マップタブの日付セレクタ用。measurement_date の一覧を返す。
  Future<List<MeasurementDate>> fetchMeasurementDates(int farmId) async {
    final res = await _dio.get('/api/farms/$farmId/results/dates');
    final list = res.data['data'] as List;
    return list
        .map((e) => MeasurementDate.fromJson(e as Map<String, dynamic>))
        .toList();
  }

  /// GET /api/farms/{farmId}/results/map?parameter=CEC&date=YYYY-MM-DD
  /// マップタブのピン描画用。測定点ごとの緯度経度 + 値を返す。
  ///
  /// [parameter]: 'CEC' | 'CaO' | 'K2O' | 'MgO'
  /// [date]: 表示する測定日 "YYYY-MM-DD"（省略時は最新日）
  Future<List<MapPoint>> fetchMapPoints(
      int farmId, {
      String parameter = 'CEC',
      String? date,
  }) async {
    final params = <String, dynamic>{'parameter': parameter};
    if (date != null) params['date'] = date;

    final res = await _dio.get(
      '/api/farms/$farmId/results/map',
      queryParameters: params,
    );
    final list = res.data['data'] as List;
    return list
        .map((e) => MapPoint.fromJson(e as Map<String, dynamic>))
        .toList();
  }
_dio の変数名と res.data のラッパーキー名について。 既存クライアントの Dio インスタンス変数名（_dio / dio / client 等）と、 レスポンスのトップレベルキー（data / points / ラッパーなし等）は 実際のコードに合わせてください。 grep -n "dio\." lib/api/farm_api_client.dart | head -10 で確認できます。

FarmDetailProvider を新規作成

lib/providers/farm_detail_provider.dart（新規作成）

圃場詳細画面（3タブ）で共通に使う状態を管理します。 タブを切り替えてもデータがリセットされないよう、 FarmDetailScreen の親ウィジェットで ChangeNotifierProvider として提供します。 

lib/providers/farm_detail_provider.dartimport 'package:flutter/foundation.dart';
import '../models/map_point.dart';
import '../models/measurement_date.dart';
// 既存クライアントのパスに合わせること
import '../api/farm_api_client.dart';

class FarmDetailProvider extends ChangeNotifier {
  final int farmId;
  final FarmApiClient _client;

  FarmDetailProvider({required this.farmId, required FarmApiClient client})
      : _client = client;

  // ── マップタブ用の状態 ────────────────────────────────────────
  List<MeasurementDate> measurementDates = [];
  String?               selectedDate;
  String                selectedParam = 'CEC'; // CEC | CaO | K2O | MgO
  List<MapPoint>         mapPoints     = [];
  bool                  isLoadingMap  = false;
  String?               mapError;

  // ── 初回ロード ─────────────────────────────────────────────
  Future<void> init() async {
    await loadMeasurementDates();
  }

  /// 測定日一覧を取得し、最新日を selectedDate にセットしてマップを読み込む。
  Future<void> loadMeasurementDates() async {
    try {
      measurementDates = await _client.fetchMeasurementDates(farmId);
      if (measurementDates.isNotEmpty) {
        selectedDate = measurementDates.first.date; // 最新日（降順前提）
        await loadMapPoints();
      }
    } catch (e) {
      mapError = e.toString();
      notifyListeners();
    }
  }

  /// パラメータまたは日付変更時にマップデータを再取得する。
  Future<void> loadMapPoints() async {
    isLoadingMap = true;
    mapError     = null;
    notifyListeners();
    try {
      mapPoints = await _client.fetchMapPoints(
        farmId,
        parameter: selectedParam,
        date:      selectedDate,
      );
    } catch (e) {
      mapError = e.toString();
    } finally {
      isLoadingMap = false;
      notifyListeners();
    }
  }

  /// パラメータタブ切替（CEC → CaO → K2O → MgO）
  Future<void> selectParam(String param) async {
    if (selectedParam == param) return;
    selectedParam = param;
    await loadMapPoints();
  }

  /// 日付セレクタで日付変更
  Future<void> selectDate(String date) async {
    if (selectedDate == date) return;
    selectedDate = date;
    await loadMapPoints();
  }
}

class _FarmListScreenState extends State<FarmListScreen> {
  final _searchCtrl = TextEditingController();
  String _query = '';

  @override
  void initState() {
    super.initState();
    // 画面表示時に圃場一覧を取得（既存 Provider のメソッドを使う）
    WidgetsBinding.instance.addPostFrameCallback((_) {
      context.read<FarmListProvider>().fetchFarms();
    });
  }

  @override
  void dispose() { _searchCtrl.dispose(); super.dispose(); }

  @override
  Widget build(BuildContext context) {
    final prov  = context.watch<FarmListProvider>();
    final farms = prov.farms
        .where((f) =>
            _query.isEmpty ||
            f.farmName.contains(_query) ||
            (f.cropType?.contains(_query) ?? false))
        .toList();

    return Scaffold(
      appBar: AppBar(
        title: const Text('圃場'),
        backgroundColor: const Color(0xFF2E5C39),
        foregroundColor: Colors.white,
      ),
      body: Column(children: [
        // 検索バー
        Padding(
          padding: const EdgeInsets.all(12),
          child: TextField(
            controller: _searchCtrl,
            decoration: const InputDecoration(
              hintText: '圃場を検索',
              prefixIcon: Icon(Icons.search),
              border: OutlineInputBorder(),
              isDense: true,
            ),
            onChanged: (v) => setState(() => _query = v),
          ),
        ),
        // 圃場一覧
        Expanded(
          child: prov.isLoading
              ? const Center(child: CircularProgressIndicator())
              : farms.isEmpty
                  ? const Center(child: Text('圃場がありません'))
                  : ListView.separated(
                      itemCount:     farms.length,
                      separatorBuilder: (_, __) => const Divider(height: 1),
                      itemBuilder:   (_, i) => _FarmCard(farm: farms[i]),
                    ),
        ),
      ]),
      // 新規圃場登録ボタン
      bottomNavigationBar: SafeArea(
        child: Padding(
          padding: const EdgeInsets.all(12),
          child: ElevatedButton.icon(
            icon:  const Icon(Icons.add),
            label: const Text('新規圃場を登録'),
            style: ElevatedButton.styleFrom(

          ),
        ),
      ),
    );

class FarmMapTab extends StatelessWidget {
  const FarmMapTab({super.key});

  static const _params = ['CEC', 'CaO', 'K2O', 'MgO'];

  @override
  Widget build(BuildContext context) {
    final prov = context.watch<FarmDetailProvider>();

    return Column(children: [
      // ── パラメータ切替チップ行 ─────────────────────────────
      Padding(
        padding: const EdgeInsets.symmetric(horizontal: 12, vertical: 8),
        child: SingleChildScrollView(
          scrollDirection: Axis.horizontal,
          child: Row(
            children: _params.map((p) => Padding(
              padding: const EdgeInsets.only(right: 8),
              child: ChoiceChip(
                label: Text(p),
                selected: prov.selectedParam == p,
                selectedColor: const Color(0xFF2E5C39),
                labelStyle: TextStyle(
                  color: prov.selectedParam == p
                      ? Colors.white : Colors.black87,
                ),
                onSelected: (_) => prov.selectParam(p),
              ),
            )).toList(),
          ),
        ),
      ),

      // ── 日付セレクタ ──────────────────────────────────────
      if (prov.measurementDates.isNotEmpty)
        Padding(
          padding: const EdgeInsets.symmetric(horizontal: 12),
          child: DropdownButton<String>(
            value: prov.selectedDate,
            isExpanded: true,
            items: prov.measurementDates.map((d) =>
              DropdownMenuItem(
                value: d.date,
                child: Text('${d.date}（${d.count}点）'),
              )
            ).toList(),
            onChanged: (v) => v != null ? prov.selectDate(v) : null,
          ),
        ),

      // ── 地図エリア ────────────────────────────────────────
      Expanded(
        child: prov.isLoadingMap
            ? const Center(child: CircularProgressIndicator())
            : prov.mapError != null
                ? Center(child: Text('エラー: ${prov.mapError}'))
                : _MapView(points: prov.mapPoints),
      ),

      // ── 下部: 統計サマリー ────────────────────────────────
      if (prov.mapPoints.isNotEmpty)
        _buildSummaryBar(prov),
    ]);
  }

  Widget _buildSummaryBar(FarmDetailProvider prov) {
    final values = prov.mapPoints
        .map((p) => p.value)
        .whereType<double>()
        .toList();
    if (values.isEmpty) return const SizedBox.shrink();
    final minVal = values.reduce(min);
    final maxVal = values.reduce(max);
    final avg    = values.reduce((a, b) => a + b) / values.length;
    final unit   = prov.mapPoints.first.unit ?? '';

    return Container(
      padding: const EdgeInsets.symmetric(horizontal: 16, vertical: 8),
      color: const Color(0xFFF5F5F0),
      child: Row(children: [
        Text('低 ${minVal.toStringAsFixed(1)}',
            style: const TextStyle(fontSize: 12)),
        Expanded(child: Container(
          height: 8, margin: const EdgeInsets.symmetric(horizontal: 8),
          decoration: BoxDecoration(
            gradient: const LinearGradient(
              colors: [Colors.blue, Colors.green, Colors.red]),
            borderRadius: BorderRadius.circular(4),
          ),
        )),
        Text('高 ${maxVal.toStringAsFixed(1)}',
            style: const TextStyle(fontSize: 12)),
        const SizedBox(width: 16),
        Text('Ave: ${avg.toStringAsFixed(1)} $unit',
            style: const TextStyle(fontWeight: FontWeight.bold, fontSize: 13)),
      ]),
    );
  }
}

// ── Google Maps ビュー ────────────────────────────────────────
class _MapView extends StatelessWidget {
  const _MapView({required this.points});
  final List<MapPoint> points;

  @override
  Widget build(BuildContext context) {
    if (points.isEmpty) {
      return const Center(child: Text('この日の測定データがありません'));
    }

    final values = points.map((p) => p.value).whereType<double>().toList();
    final minV   = values.isEmpty ? 0.0 : values.reduce(min);
    final maxV   = values.isEmpty ? 1.0 : values.reduce(max);

    final initialPos = LatLng(points.first.latitude, points.first.longitude);

    return GoogleMap(
      initialCameraPosition: CameraPosition(target: initialPos, zoom: 17),
      mapType: MapType.satellite,
      myLocationButtonEnabled: false,
      markers: points.indexed.map(((i, p)) {
        final ratio = (maxV == minV) ? 0.5
            : ((p.value ?? minV) - minV) / (maxV - minV);
        final hue = ui.lerpDouble(240, 0, ratio)!; // 青(240) → 赤(0)
        return Marker(
          markerId: MarkerId('pin_$i'),
          position: LatLng(p.latitude, p.longitude),
          icon: BitmapDescriptor.defaultMarkerWithHue(hue.clamp(0, 360).toDouble()),
          infoWindow: InfoWindow(
            title: '${i + 1}: ${p.value?.toStringAsFixed(1) ?? "—"} ${p.unit ?? ""}',
          ),
        );
      }).toSet(),
    );
  }
}
カスタム番号付き円マーカーについて。 上記は BitmapDescriptor.defaultMarkerWithHue で色分けした標準ピンを使います。 仕様書の「数値ラベル付きの円マーカー」を完全再現するには Canvas で描画したカスタムアイコンが必要です（処理コスト高）。 Phase 4 では標準ピン + InfoWindow で実データ確認を優先し、 カスタム円マーカーは Phase 7（測定画面刷新）合わせで対応する方針にしています。 Cursor に完全なカスタムマーカーを実装させたい場合は、 この callout を Cursor に見せて判断させてください。
