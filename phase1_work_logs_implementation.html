        });

    public function down(): void

ローカルで migrate を実行して SQL と構造を確認

ターミナル / ローカル
# まず --pretend で生成される SQL を目視確認（実際には何も変更しない） $ docker compose exec php-fpm php artisan migrate --pretend # CREATE TABLE `work_logs` ... の SQL が表示されれば OK # ローカル DB に対して実際に migrate を実行 $ docker compose exec php-fpm php artisan migrate # INFO Running migrations. と表示されテーブルが作成される # カラム構造を確認 $ docker compose exec php-fpm php artisan db:table work_logs # farm_id / work_type / work_date / title / ... が一覧表示されれば成功

本番 RDS への migrate は CI/CD が行います。 ここでのローカル migrate はあくまで SQL の確認とテスト実行用です。 EC2 に SSH して手動で migrate コマンドを叩く必要はありません。 main ブランチへのマージ後、GitHub Actions が自動的に php artisan migrate --force を本番 RDS に対して実行します。

app/Policies/WorkLogPolicy.php<?php

namespace App\Policies;

use App\Models\AppUser;
use App\Models\Farm;
use App\Models\WorkLog;

class WorkLogPolicy
{
    /**
     * 作業ログ一覧の閲覧は、その圃場の所有者のみ許可。
     */
    public function viewAny(AppUser $user, Farm $farm): bool
    {
        return $farm->app_user_id === $user->id;
    }

    /**
     * 作業ログ作成は、その圃場の所有者のみ許可。
     */
    public function create(AppUser $user, Farm $farm): bool
    {
        return $farm->app_user_id === $user->id;
    }

    /**
     * 作業ログ更新・削除は、その WorkLog が属する圃場の所有者のみ許可。
     */
    public function update(AppUser $user, WorkLog $workLog): bool
    {
        return $workLog->farm->app_user_id === $user->id;
    }

    public function delete(AppUser $user, WorkLog $workLog): bool
    {
        return $workLog->farm->app_user_id === $user->id;
    }
}
AppUser について。 モバイル API の認証ユーザーは App\Models\AppUser です（App\Models\User とは別物）。 Cognito JWT ミドルウェアが auth()->user() に AppUser をセットしている前提で書いています。 既存の認証ミドルウェアの実装に応じて型を調整してください。

            'detail'       => ['nullable', 'string'],
            'amount_value' => ['nullable', 'numeric', 'min:0'],
            'amount_unit'  => ['nullable', 'string', 'max:16'],
            'scope'        => ['nullable', 'in:whole,partial'],
        ];
    }
}

WorkLogController を新規作成

app/Http/Controllers/Api/WorkLogController.php（新規作成） / ローカル

app/Http/Controllers/Api/WorkLogController.php<?php

namespace App\Http\Controllers\Api;

use App\Http\Controllers\Controller;
use App\Http\Requests\StoreWorkLogRequest;
use App\Http\Requests\UpdateWorkLogRequest;
use App\Models\Farm;
use App\Models\WorkLog;
use Illuminate\Http\JsonResponse;

class WorkLogController extends Controller
{
    /**
     * GET /api/v1/farms/{farm}/work-logs
     * 圃場に紐づく作業ログ一覧（work_date 降順）
     */
    public function index(Farm $farm): JsonResponse
    {
        $this->authorize('viewAny', [WorkLog::class, $farm]);

        $logs = $farm->workLogs()->get();

        return response()->json(['data' => $logs]);
    }

    /**
     * POST /api/v1/farms/{farm}/work-logs
     * 作業ログ作成
     */
    public function store(
        StoreWorkLogRequest $request,
        Farm $farm
    ): JsonResponse {
        $this->authorize('create', [WorkLog::class, $farm]);

        $log = $farm->workLogs()->create($request->validated());

        return response()->json(['data' => $log], 201);
    }

    /**
     * PATCH /api/v1/work-logs/{workLog}
     * 作業ログ更新
     */
    public function update(
        UpdateWorkLogRequest $request,
        WorkLog $workLog
    ): JsonResponse {
        $this->authorize('update', $workLog);

        $workLog->update($request->validated());

        return response()->json(['data' => $workLog->fresh()]);
    }

    /**
     * DELETE /api/v1/work-logs/{workLog}
     * 作業ログ削除
     */
    public function destroy(WorkLog $workLog): JsonResponse
    {
        $this->authorize('delete', $workLog);

        $workLog->delete();

        return response()->json(null, 204);
    }
}

tests/Feature/WorkLogApiTest.php<?php

namespace Tests\Feature;

use App\Models\AppUser;
use App\Models\Farm;
use App\Models\WorkLog;
use Illuminate\Foundation\Testing\RefreshDatabase;
use Tests\TestCase;

class WorkLogApiTest extends TestCase
{
    use RefreshDatabase;

    // ─── ヘルパー ─────────────────────────────────────

    /**
     * AppUser を作成して認証済みの状態にする。
     * ※既存プロジェクトに認証ヘルパーがある場合はそちらを使用。
     */
    private function authAs(AppUser $user): static
    {
        // 例: JWT を使っているなら actingAs($user, 'api') など
        // 既存テストの認証方式に合わせて変更してください
        return $this->actingAs($user, 'api');
    }

    private function makeUserWithFarm(): array
    {
        $user = AppUser::factory()->create();
        $farm = Farm::factory()->create(['app_user_id' => $user->id]);
        return [$user, $farm];
    }

    private function validPayload(): array
    {
        return [
            'work_type'    => 'fertilization',
            'work_date'    => '2026-03-16',
            'title'        => 'NK化成 20kg',
            'amount_value' => 20.0,
            'amount_unit'  => 'kg',
        ];
    }

    // ─── 認証・認可 ───────────────────────────────────

    public function test_unauthenticated_request_returns_401(): void
    {
        [, $farm] = $this->makeUserWithFarm();

        $this->getJson("/api/v1/farms/{$farm->id}/work-logs")
             ->assertStatus(401);
    }

    public function test_another_users_farm_returns_403(): void
    {
        [$owner, $farm] = $this->makeUserWithFarm();
        $other = AppUser::factory()->create();

        $this->authAs($other)
             ->getJson("/api/v1/farms/{$farm->id}/work-logs")
             ->assertStatus(403);
    }

    // ─── 一覧 ─────────────────────────────────────────

    public function test_index_returns_own_work_logs(): void
    {
        [$user, $farm] = $this->makeUserWithFarm();
        WorkLog::factory()->count(3)->create(['farm_id' => $farm->id]);

        $this->authAs($user)
             ->getJson("/api/v1/farms/{$farm->id}/work-logs")
             ->assertStatus(200)
             ->assertJsonCount(3, 'data');
    }

    // ─── 作成 ─────────────────────────────────────────

    public function test_store_creates_work_log(): void
    {
        [$user, $farm] = $this->makeUserWithFarm();

        $this->authAs($user)
             ->postJson("/api/v1/farms/{$farm->id}/work-logs", $this->validPayload())
             ->assertStatus(201)
             ->assertJsonPath('data.work_type', 'fertilization')
             ->assertJsonPath('data.title', 'NK化成 20kg');

        $this->assertDatabaseHas('work_logs', [
            'farm_id'   => $farm->id,
            'work_type' => 'fertilization',
        ]);
    }

    public function test_store_fails_with_invalid_work_type(): void
    {
        [$user, $farm] = $this->makeUserWithFarm();
        $payload = $this->validPayload();
        $payload['work_type'] = 'invalid_type';

        $this->authAs($user)
             ->postJson("/api/v1/farms/{$farm->id}/work-logs", $payload)
             ->assertStatus(422)
             ->assertJsonValidationErrors(['work_type']);
    }

    public function test_store_fails_with_missing_required_fields(): void
    {
        [$user, $farm] = $this->makeUserWithFarm();

             ->postJson("/api/v1/farms/{$farm->id}/work-logs", [])

    public function test_update_modifies_work_log(): void

    public function test_destroy_deletes_work_log(): void

curl スモークテスト（本番エンドポイント）

ターミナル / ローカル端末（SSH から抜けて実施）

有効な Cognito アクセストークンと、自分が保有している圃場の ID を使って確認します。 テストで作成したデータは最後に DELETE で後片付けします。

# 環境変数をセット $ export BASE="https://api.hm-admin.com" $ export TOKEN="eyJhbGciOi..." # Cognito から発行した access_token $ export FARM_ID=12 # 自分が保有している圃場 ID
# ① 認証なし → 401 が返ること $ curl -s -o /dev/null -w "%{http_code}\n" \ "$BASE/api/v1/farms/$FARM_ID/work-logs" # 401 と表示されること
# ② 作成（POST） → 201 + data が返ること $ curl -s -X POST "$BASE/api/v1/farms/$FARM_ID/work-logs" \ -H "Authorization: Bearer $TOKEN" \ -H "Content-Type: application/json" \ -d '{ "work_type": "fertilization", "work_date": "2026-05-14", "title": "スモークテスト用データ", "amount_value": 10.0, "amount_unit": "kg" }' | jq '{status: "201 expected", id: .data.id, type: .data.work_type}' # {"status": "201 expected", "id": X, "type": "fertilization"} が返れば OK # 返ってきた id を保存 $ export WORK_LOG_ID=<上で返ってきた id>
# ③ 一覧取得（GET） → 200 + data 配列が返ること $ curl -s "$BASE/api/v1/farms/$FARM_ID/work-logs" \ -H "Authorization: Bearer $TOKEN" | jq '.data | length' # 1 以上の数字が返ること
# ④ 更新（PATCH） → 200 + title が変わること $ curl -s -X PATCH "$BASE/api/v1/work-logs/$WORK_LOG_ID" \ -H "Authorization: Bearer $TOKEN" \ -H "Content-Type: application/json" \ -d '{"title": "スモークテスト完了"}' | jq '.data.title' # "スモークテスト完了" が返ること
# ⑤ 削除（DELETE） → 204 が返ること（テストデータの後片付け） $ curl -s -o /dev/null -w "%{http_code}\n" \ -X DELETE "$BASE/api/v1/work-logs/$WORK_LOG_ID" \ -H "Authorization: Bearer $TOKEN" # 204 と表示されること
5項目すべて OK であれば Phase 1 完了です。 削除した後に GET でも 0 件または正しい件数が返ることを念のため確認してから終了してください。
