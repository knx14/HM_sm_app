  final String    workDate;   // "YYYY-MM-DD"
  final String?   title;
  final String?   detail;
  final double?   amountValue;
  final String?   amountUnit;
  final String    scope;      // 'whole'（Phase 6 は全体固定）

    required this.workDate,
    this.title,

    this.amountValue,

    'work_date':    workDate,

    if (amountValue  != null) 'amount_value': amountValue,
    if (amountUnit   != null) 'amount_unit':  amountUnit,

    workDate:    json['work_date']    as String,

オフライン時の作業ログを pending_work_logs.json に保存します。 PendingUploadStore の実装を参考に同じパターンで書いてください。 以下は骨格です。 

lib/features/work_logs/data/work_log_queue.dartimport 'dart:convert';
import 'dart:io';
import 'package:path_provider/path_provider.dart';
import '../domain/work_log_entry.dart';

/// オフライン時の作業ログを pending_work_logs.json に保存・取得するクラス。
/// PendingUploadStore と同じ JSON ファイル方式で実装する。
class WorkLogQueue {
  static const _fileName = 'pending_work_logs.json';

  // ── ファイルパス取得 ──────────────────────────────────────────
  Future<File> _file() async {
    final dir = await getApplicationDocumentsDirectory();
    return File('${dir.path}/$_fileName');
  }

  /// キュー全件取得
  Future<List<PendingWorkLog>> readAll() async {
    final f = await _file();
    if (!await f.exists()) return [];
    final raw  = await f.readAsString();
    final list = jsonDecode(raw) as List;
    return list
        .map((e) => PendingWorkLog.fromJson(e as Map<String, dynamic>))
        .toList();
  }

  /// キューに1件追加
  Future<void> enqueue(int farmId, WorkLogEntry entry) async {
    final items = await readAll();
    items.add(PendingWorkLog(
      localId:   '${DateTime.now().millisecondsSinceEpoch}',
      farmId:    farmId,
      entry:     entry,
      createdAt: DateTime.now().toIso8601String(),
    ));
    await _write(items);
  }

  /// 送信成功後に localId を指定して削除
  Future<void> remove(String localId) async {
    final items = await readAll();
    items.removeWhere((i) => i.localId == localId);
    await _write(items);
  }

  /// キューを全件削除
  Future<void> clear() async {
    final f = await _file();
    if (await f.exists()) await f.writeAsString('[]');
  }

  Future<void> _write(List<PendingWorkLog> items) async {
    final f = await _file();
    await f.writeAsString(jsonEncode(items.map((i) => i.toJson()).toList()));
  }
}

/// キュー内の1アイテム（localId + farmId + entry）
class PendingWorkLog {
  final String       localId;    // タイムスタンプ由来の一意ID
  final int          farmId;
  final WorkLogEntry entry;
  final String       createdAt;

  const PendingWorkLog({
    required this.localId,
    required this.farmId,
    required this.entry,
    required this.createdAt,
  });

  Map<String, dynamic> toJson() => {
    'local_id':   localId,
    'farm_id':    farmId,
    'entry':      entry.toJson(),
    'created_at': createdAt,
  };

  factory PendingWorkLog.fromJson(Map<String, dynamic> j) => PendingWorkLog(
    localId:   j['local_id']   as String,
    farmId:    j['farm_id']    as int,
    entry:     WorkLogEntry.fromJson(j['entry'] as Map<String, dynamic>),
    createdAt: j['created_at'] as String,
  );
}

WorkLogRepository を作成（オンライン API 呼び出し + キュー統合）

lib/features/work_logs/data/work_log_repository.dart（新規作成） 
lib/features/work_logs/data/work_log_repository.dartimport 'package:connectivity_plus/connectivity_plus.dart';
import 'package:dio/dio.dart';
import '../domain/work_log_entry.dart';
import 'work_log_queue.dart';

/// 作業ログの CRUD + オフラインキューを一元管理するリポジトリ。
///
/// オンライン時: Dio で API を直接呼ぶ。
/// オフライン時: WorkLogQueue（JSON ファイル）に追加し、
///              ConnectivityPlus で復帰を検知してリトライする。
class WorkLogRepository {
  final Dio          _dio;
  final WorkLogQueue  _queue;

  WorkLogRepository({required Dio dio, WorkLogQueue? queue})
      : _dio = dio, _queue = queue ?? WorkLogQueue();

  // ── 作成 ─────────────────────────────────────────────────────

  /// POST /api/v1/farms/{farmId}/work-logs
  ///
  /// オンラインなら即座に API 送信。
  /// オフライン（[DioException] 503/接続不可）なら JSON キューに追加して [true] を返す。
  /// API バリデーションエラー (422) はそのままスロー。
  Future<WorkLogSaveResult> create(int farmId, WorkLogEntry entry) async {
    // ネットワーク接続確認
    final conn = await Connectivity().checkConnectivity();
    final isOnline = conn != ConnectivityResult.none;

    if (!isOnline) {
      await _queue.enqueue(farmId, entry);
      return const WorkLogSaveResult(queued: true);
    }

    try {
      final res = await _dio.post(
        '/api/v1/farms/$farmId/work-logs',
        data: entry.toJson(),
      );
      return WorkLogSaveResult(
        queued: false,
        serverId: res.data['data']['id'] as int?,
      );
    } on DioException catch (e) {
      // タイムアウト・接続切断 → キューに積む
      if (_isOfflineError(e)) {
        await _queue.enqueue(farmId, entry);
        return const WorkLogSaveResult(queued: true);
      }
      rethrow; // 422 バリデーションエラー等はそのままスロー
    }
  }

  // ── オフラインキュー処理 ───────────────────────────────────────

  /// キュー内の全 PendingWorkLog を順次 API 送信し、成功したものを削除する。
  /// ConnectivityPlus の復帰イベントから呼ばれる。
  Future<void> flushQueue() async {
    final items = await _queue.readAll();
    for (final pending in items) {
      try {
        await _dio.post(
          '/api/v1/farms/${pending.farmId}/work-logs',
          data: pending.entry.toJson(),
        );
        await _queue.remove(pending.localId);
      } on DioException catch (_) {
        // 1件失敗しても次のアイテムを続ける
        continue;
      }
    }
  }

  /// キューの件数を返す（同期ボタンバッジ連携用）
  Future<int> pendingCount() async {
    return (await _queue.readAll()).length;
  }

  bool _isOfflineError(DioException e) =>
      e.type == DioExceptionType.connectionTimeout ||
      e.type == DioExceptionType.connectionError  ||
      e.type == DioExceptionType.receiveTimeout;
}

/// create() の返り値。
/// queued=true はオフラインキューに積まれたことを示す。
class WorkLogSaveResult {
  final bool queued;
  final int?  serverId;
  const WorkLogSaveResult({required this.queued, this.serverId});
}

lib/features/work_logs/presentation/work_log_notifier.dartimport 'package:flutter/foundation.dart';

/// WorkLogEditScreen の状態管理。
/// フォームの3ステップ状態を保持し、保存処理を担う。
class WorkLogNotifier extends ChangeNotifier {
  final int                _farmId;
  final WorkLogRepository _repo;

  WorkLogNotifier({required int farmId, required WorkLogRepository repo})
      : _farmId = farmId, _repo = repo;

  // ── フォーム状態 ──────────────────────────────────────────────
  int        step       = 0;             // 0=種別選択 1=詳細入力 2=確認
  WorkType?  workType;
  String     workDate   = '';            // "YYYY-MM-DD"
  String     title      = '';
  String     detail     = '';
  String     amountText = '';            // "20.0" 等
  String     amountUnit = '';

  // ── 保存状態 ──────────────────────────────────────────────────
  bool    isSaving  = false;
  bool?   saveQueued;   // null=未完了 true=キューに追加 false=API送信済み
  String? saveError;

  // ── ステップ操作 ──────────────────────────────────────────────
  void selectWorkType(WorkType t) {
    workType = t;
    // 今日の日付をデフォルトにセット
    if (workDate.isEmpty) {
      final now = DateTime.now();
      workDate = '${now.year}-${now.month.toString().padLeft(2, "0")}'
                 '-${now.day.toString().padLeft(2, "0")}';
    }
    step = 1;
    notifyListeners();
  }

  void goToConfirm({
    required String newTitle,
    required String newDetail,
    required String newDate,
    required String newAmountText,
    required String newAmountUnit,
  }) {
    title      = newTitle;
    detail     = newDetail;
    workDate   = newDate;
    amountText = newAmountText;
    amountUnit = newAmountUnit;
    step       = 2;
    notifyListeners();
  }

  void goBack() {
    if (step > 0) { step--; notifyListeners(); }
  }

  // ── 保存 ─────────────────────────────────────────────────────
  Future<void> save() async {
    if (workType == null) return;
    isSaving  = true;
    saveError = null;
    notifyListeners();

    final entry = WorkLogEntry(
      workType:    workType!.value,
      workDate:    workDate,
      title:       title.isNotEmpty    ? title    : null,
      detail:      detail.isNotEmpty   ? detail   : null,
      amountValue: double.tryParse(amountText),
      amountUnit:  amountUnit.isNotEmpty ? amountUnit : null,
    );

    try {
      final result = await _repo.create(_farmId, entry);
      saveQueued = result.queued;
    } catch (e) {
      saveError = e.toString();
    } finally {
      isSaving = false;
      notifyListeners();
    }
  }

  bool get isSaveComplete => saveQueued != null && saveError == null;
}

class _StepIndicator extends StatelessWidget {

  Widget build(BuildContext context) {

        final isDone   = i < currentStep;
        return Row(children: [
          CircleAvatar(
            radius: 12,
            backgroundColor: isDone || isActive
                ? const Color(0xFF2E5C39) : Colors.grey[300],

                style: TextStyle(color: isDone || isActive ? Colors.white : Colors.grey, fontSize: 11)),
          ),
          Text(' $label', style: TextStyle(fontSize: 12, color: isActive ? Colors.black87 : Colors.grey)),
          if (i < labels.length - 1) Container(width: 20, height: 1,
              margin: const EdgeInsets.symmetric(horizontal: 6),
              color: Colors.grey[300]),
        ]);
      }).toList(),
